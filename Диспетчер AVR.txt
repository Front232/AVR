#include <avr/io.h>                 
#include <avr/interrupt.h>          
#include <util/atomic.h>            

////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////

#define MAX_TASKS 20                                                           // Определяем максимальное количество задач в списке
#define MAX 0                                                                  // Определяем константу для бесконечного выполнения

typedef enum
{                                                                              // Создаем перечисление для уровней приоритета
	HIGH,                           
	MED,                            
	LOW                             
} en;                               

typedef struct
{                    
	void (*f)(void);            
	uint16_t delay;                                                              // Задержка перед первым или следующим запуском (мс)
	uint16_t period;                                                             // Период повторения задачи (мс)
	uint16_t cnt;                                                                // Счетчик количества запусков (0 - бесконечно)
	uint8_t prioritet;                                                           // Приоритет выполнения задачи (0 - самый важный)
	uint8_t run;                                                                 // Флаг готовности задачи к исполнению (1 - пора)
} task;                             

volatile task task_mass[MAX_TASKS];                                            // Создаем массив задач, доступный для прерываний

///////  НАСТРОЙКИ ОЧЕРЕДИ КНОПОК (FIFO)  //////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////

#define KEY_BUF_SIZE 16                                                        // Определяем размер буфера кодов нажатых кнопок
volatile char key_buffer[KEY_BUF_SIZE];                         
volatile uint8_t golova = 0;                                                   // Индекс для записи новых данных в голову очереди
volatile uint8_t hvost = 0;                                                    // Индекс для чтения старых данных из хвоста очереди

///////  НАСТРОЙКИ УНИВЕРСАЛЬНЫХ КНОПОК  ///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////

typedef struct
{                                                                             // Описываем структуру настроек кнопки
	volatile uint8_t *port;                                                     // Указатель на регистр порта ввода (например, &PINA)
	uint8_t pin_bit;                                                            // Номер бита соответствующего пина (0..7)
	uint8_t active_level;                                                       // Уровень сигнала при нажатии (0 - GND, 1 - VCC)
	char code;                                                                  // Символ-код, возвращаемый при нажатии
	uint16_t counter;                                                           // Текущий счетчик времени нажатого состояния
	uint16_t limit;                                                             // Порог времени для защиты от дребезга (мс)
	uint16_t cnt_avto_povtor;                                                   // Счетчик интервала автоповтора
	uint16_t period_povtorov;                                                   // Скорость (период) генерации повторов (мс)
	uint8_t fl_otpuskanie_key;                                                  // Флаг, сигнализирующий об отпускании кнопки
} Button;

#define NUM_KEYS 3                                                            // Задаем общее количество кнопок в системе

Button key[NUM_KEYS] =
{                                         
	{&PINA, PA0, 0, 'u', 0, 50, 0, 150, 1},                                     // Кнопка UP на PINA0, активный 0, повтор 150мс
	{&PINA, PA1, 0, 'd', 0, 50, 0, 150, 1},                                     // Кнопка DOWN на PINA1, активный 0, повтор 150мс
	{&PINC, PC0, 1, 's', 0, 2000, 0, 0, 1}                                      // Кнопка START на PINC0, активный 1, удержание 2с
};                                  

///////  ФУНКЦИИ БУФЕРА КНОПОК  ////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////

void Write_key(char code)                                                     // Функция добавления кода в очередь
{                                  
	uint8_t next = (golova + 1) % KEY_BUF_SIZE;                                 // Вычисляем индекс следующей ячейки
	
	if (next != hvost)                                                          // Проверяем, есть ли свободное место в буфере
	{
		key_buffer[golova] = code;                                                // Записываем код в текущую свободную ячейку
		golova = next;                                                            // Сдвигаем индекс головы очереди
	}                              
}

char Read_key(void)                                                           // Функция извлечения кода из очереди
{                                
	if (golova == hvost)                                                        // Проверяем, не пуст ли буфер
	{
		return 0;                                                                 // Если голова равна хвосту, возвращаем 0
	}                              
	char code = key_buffer[hvost++];                                            // Считываем код из хвоста очереди
	hvost &=  KEY_BUF_SIZE-1;                                                   // Сдвигаем индекс хвоста очереди
	return code;                                                                // Возвращаем считанный код
}

void Clear_key_buffer(void)                                                   // Функция полной очистки буфера
{
	ATOMIC_BLOCK(ATOMIC_RESTORESTATE) {hvost = golova;}                         // Приравниваем хвост к голове (очередь становится пустой)
}

///////   ФУНКЦИИ ДИСПЕТЧЕРА  //////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////

void Init_dispetcher(void)
{                                
	TCCR0A = (1 << WGM01);                                                      // Устанавливаем режим таймера CTC (сброс при совпадении)
	TCCR0B = (1 << CS01) | (1 << CS00);                                         // Устанавливаем делитель частоты на 64
	OCR0A = 249;                                                                // Задаем порог: 16МГц / (64 * 1000Гц) - 1 = 249 (1мс)
	TIMSK0 = (1 << OCIE0A);         
}                                

////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////

ISR(TIMER0_COMPA_vect)                                                        // Обработчик прерывания таймера (каждую 1 мс)
{                               
	for (uint8_t i = 0; i < MAX_TASKS; i++)                                     // Проходим циклом по всем задачам
	{
		if (task_mass[i].f)                                                       // Если в текущей ячейке есть назначенная задача
		{
			if (task_mass[i].delay == 0)                                            // Если время ожидания задачи истекло
			{
				task_mass[i].run = 1;                                                 // Устанавливаем флаг готовности к запуску
				task_mass[i].delay = task_mass[i].period;                             // Перезаряжаем задержку периодом
			}
			else
			{                     
				task_mass[i].delay--;                                                 // Уменьшаем счетчик задержки на 1 мс
			}                       
		}                         
	}                           
}

////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////

void Add_task(void (*f)(void), uint16_t delay, uint16_t period, uint16_t count, uint8_t prio) // Функция добавления задачи
{                                     
	for (uint8_t i = 0; i < MAX_TASKS; i++)                                     // Ищем свободную ячейку в массиве
	{ 
		if (task_mass[i].f == 0)                                                  // Если ячейка пуста (указатель равен нулю)
		{ 
			task_mass[i].f = f;                                                     // Сохраняем адрес функции задачи
			task_mass[i].delay = delay;                                             // Устанавливаем начальную задержку
			task_mass[i].period = (period == 0 ? 1 : period);                       // Устанавливаем период повторения
			task_mass[i].cnt = count;                                               // Устанавливаем количество выполнений
			task_mass[i].prioritet = prio;                                          // Устанавливаем приоритет задачи
			task_mass[i].run = 0;                                                   // Сбрасываем флаг запуска
			break;              
		}                       
	}                           
}                                   

////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////

void Dispatch_tasks(void)
{                                
	while (1)  
	{                    
		int8_t Indx = -1;                                                          // Переменная для хранения индекса самой важной задачи
		uint8_t min_prio = 255;                                                    // Начальное значение для поиска минимального приоритета
		
		for (uint8_t i = 0; i < MAX_TASKS; i++)                                    // Ищем задачу с наивысшим приоритетом
		{ 
			if (task_mass[i].f && task_mass[i].run)                                  // Если задача существует и готова
			{ 
				if (task_mass[i].prioritet < min_prio)                                 // Если приоритет этой задачи выше найденного
				{ 
					min_prio = task_mass[i].prioritet;                                   // Запоминаем новый лучший приоритет
					Indx = i;                                                            // Сохраняем индекс этой задачи
				}                              
			}                       
		}                           
		
		if (Indx == -1) {break;}                                                    // Если готовых к исполнению задач не обнаружено	                       
		
		volatile task *t = &task_mass[Indx];                                        // Берем указатель на выбранную задачу		
		t->run = 0;                                                                // Сбрасываем флаг запуска (задача начала выполняться)
		t->f();                                                                    // ВЫЗЫВАЕМ ФУНКЦИЮ ЗАДАЧИ 
		
		if ((t->cnt > 0) && (--t->cnt == 0))                                       // Если счетчик был задан и он обнулился
		{ 
				t->f = 0;                                                              // Удаляем задачу из списка
		}                       
	}                             
}                                   

////////  ЗАДАЧА СКАНИРОВАНИЯ КНОПОК  ///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////

void Task_keys_scan(void)                                                      // Функция фонового сканирования кнопок
{                                         
	for (uint8_t i = 0; i < NUM_KEYS; i++) 
	{ 
		uint8_t x = (*(key[i].port) & (1 << key[i].pin_bit)) ? 1 : 0;              // Читаем бит порта
		
		if (x == key[i].active_level)                                              // Если считанный уровень совпадает с активным
		{ 
			if (key[i].counter < key[i].limit)                                       // Если время удержания меньше порога дребезга
			{ 
				key[i].counter++;                                                      // Увеличиваем счетчик времени нажатия
			}
			else 
			{                                                                        // Если кнопка удерживается дольше порога антидребезга
				if (key[i].fl_otpuskanie_key)                                          // Если это первое срабатывание после отпускания
				{ 
					Write_key(key[i].code);                                              // Кладём код кнопки в буфер очереди
					key[i].fl_otpuskanie_key = 0;                                        // Сбрасываем флаг отпускания (кнопка зажата)
					key[i].cnt_avto_povtor = 0;                                          // Обнуляем счетчик для начала паузы повтора
				} 
				else 
				{
					if (key[i].period_povtorov > 0)                                      // Если для кнопки разрешен автоповтор
					{ 
					  if (++key[i].cnt_avto_povtor >= key[i].period_povtorov)            // Если пришло время повтора
					  { 
						  Write_key(key[i].code);                                          // Кладём код повторно в буфер
						  key[i].cnt_avto_povtor = 0;                                      // Сбрасываем счетчик интервала повтора
					  }           
				  }    
				}
			}                   
		} 
		else 
		{                                                         
			key[i].counter = 0;                                                      // Обнуляем основной счетчик времени нажатия
			key[i].cnt_avto_povtor = 0;                                              // Обнуляем счетчик повторов
			key[i].fl_otpuskanie_key = 1;                                            // Устанавливаем флаг отпускания
		}                           
	}                              
}
